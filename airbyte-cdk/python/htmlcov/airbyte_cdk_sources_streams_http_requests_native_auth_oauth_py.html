6from typing import Any, List, Mapping, MutableMapping, Tuple 

7 

8import pendulum 

9import requests 

10from requests.auth import AuthBase 

11 

12 

13class Oauth2Authenticator(AuthBase): 

14 """ 

15 Generates OAuth2.0 access tokens from an OAuth2.0 refresh token and client credentials. 

16 The generated access token is attached to each request via the Authorization header. 

17 """ 

18 

19 def __init__( 

20 self, 

21 token_refresh_endpoint: str, 

22 client_id: str, 

23 client_secret: str, 

24 refresh_token: str, 

25 scopes: List[str] = None, 

26 token_expiry_date: pendulum.datetime = None, 

27 access_token_name: str = "access_token", 

28 expires_in_name: str = "expires_in", 

29 ): 

30 self.token_refresh_endpoint = token_refresh_endpoint 

31 self.client_secret = client_secret 

32 self.client_id = client_id 

33 self.refresh_token = refresh_token 

34 self.scopes = scopes 

35 self.access_token_name = access_token_name 

36 self.expires_in_name = expires_in_name 

37 

38 self._token_expiry_date = token_expiry_date or pendulum.now().subtract(days=1) 

39 self._access_token = None 

40 

41 def __call__(self, request): 

42 request.headers.update(self.get_auth_header()) 

43 return request 

44 

45 def get_auth_header(self) -> Mapping[str, Any]: 

46 return {"Authorization": f"Bearer {self.get_access_token()}"} 

47 

48 def get_access_token(self): 

49 if self.token_has_expired(): 

50 t0 = pendulum.now() 

51 token, expires_in = self.refresh_access_token() 

52 self._access_token = token 

53 self._token_expiry_date = t0.add(seconds=expires_in) 

54 

55 return self._access_token 

56 

57 def token_has_expired(self) -> bool: 

58 return pendulum.now() > self._token_expiry_date 

59 

60 def get_refresh_request_body(self) -> Mapping[str, Any]: 

61 """Override to define additional parameters""" 

62 payload: MutableMapping[str, Any] = { 

63 "grant_type": "refresh_token", 

64 "client_id": self.client_id, 

65 "client_secret": self.client_secret, 

66 "refresh_token": self.refresh_token, 

67 } 

68 

69 if self.scopes: 

70 payload["scopes"] = self.scopes 

71 

72 return payload 

73 

74 def refresh_access_token(self) -> Tuple[str, int]: 

75 """ 

76 returns a tuple of (access_token, token_lifespan_in_seconds) 

77 """ 

78 try: 

79 response = requests.request(method="POST", url=self.token_refresh_endpoint, data=self.get_refresh_request_body()) 

80 response.raise_for_status() 

81 response_json = response.json() 

82 return response_json[self.access_token_name], response_json[self.expires_in_name] 

83 except Exception as e: 

84 raise Exception(f"Error while refreshing access token: {e}") from e
