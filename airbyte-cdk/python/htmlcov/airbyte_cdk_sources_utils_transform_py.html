15class TransformConfig(Flag): 

16 """ 

17 TypeTransformer class config. Configs can be combined using bitwise or operator e.g. 

18 ``` 

19 TransformConfig.DefaultSchemaNormalization | TransformConfig.CustomSchemaNormalization 

20 ``` 

21 """ 

22 

23 # No action taken, default behaviour. Cannot be combined with any other options. 

24 NoTransform = auto() 

25 # Applies default type casting with default_convert method which converts 

26 # values by applying simple type casting to specified jsonschema type. 

27 DefaultSchemaNormalization = auto() 

28 # Allow registering custom type transformation callback. Can be combined 

29 # with DefaultSchemaNormalization. In this case default type casting would 

30 # be applied before custom one. 

31 CustomSchemaNormalization = auto()

36 Class for transforming object before output.

41 def __init__(self, config: TransformConfig): 

42 """ 

43 Initialize TypeTransformer instance. 

44 :param config Transform config that would be applied to object 

45 """ 

46 if TransformConfig.NoTransform in config and config != TransformConfig.NoTransform: 

47 raise Exception("NoTransform option cannot be combined with other flags.") 

48 self._config = config 

49 all_validators = { 

50 key: self.__get_normalizer(key, orig_validator) 

51 for key, orig_validator in Draft7Validator.VALIDATORS.items() 

52 # Do not validate field we do not transform for maximum performance. 

53 if key in ["type", "array", "$ref", "properties", "items"] 

54 } 

55 self._normalizer = validators.create(meta_schema=Draft7Validator.META_SCHEMA, validators=all_validators)

57 def registerCustomTransform(self, normalization_callback: Callable[[Any, Dict[str, Any]], Any]) -> Callable: 

58 """ 

59 Register custom normalization callback. 

60 :param normalization_callback function to be used for value 

61 normalization. Takes original value and part type schema. Should return 

62 normalized value. See docs/connector-development/cdk-python/schemas.md 

63 for details. 

64 :return Same callbeck, this is usefull for using registerCustomTransform function as decorator. 

65 """ 

66 if TransformConfig.CustomSchemaNormalization not in self._config: 

67 raise Exception("Please set TransformConfig.CustomSchemaNormalization config before registering custom normalizer") 

68 self._custom_normalizer = normalization_callback 

69 return normalization_callback

71 def __normalize(self, original_item: Any, subschema: Dict[str, Any]) -> Any: 

72 """ 

73 Applies different transform function to object's field according to config. 

74 :param original_item original value of field. 

75 :param subschema part of the jsonschema containing field type/format data. 

76 :return Final field value. 

77 """ 

78 if TransformConfig.DefaultSchemaNormalization in self._config: 

79 original_item = self.default_convert(original_item, subschema) 

80 

81 if self._custom_normalizer: 

82 original_item = self._custom_normalizer(original_item, subschema) 

83 return original_item

85 @staticmethod 

86 def default_convert(original_item: Any, subschema: Dict[str, Any]) -> Any: 

87 """ 

88 Default transform function that is used when TransformConfig.DefaultSchemaNormalization flag set. 

89 :param original_item original value of field. 

90 :param subschema part of the jsonschema containing field type/format data. 

91 :return transformed field value. 

92 """ 

93 target_type = subschema.get("type") 

94 if original_item is None and "null" in target_type: 

95 return None 

96 if isinstance(target_type, list): 

97 # jsonschema type could either be a single string or array of type 

98 # strings. In case if there is some disambigous and more than one 

99 # type (except null) do not do any conversion and return original 

100 # value. If type array has one type and null i.e. {"type": 

101 # ["integer", "null"]}, convert value to specified type. 

102 target_type = [t for t in target_type if t != "null"] 

103 if len(target_type) != 1: 

104 return original_item 

105 target_type = target_type[0] 

106 try: 

107 if target_type == "string": 

108 return str(original_item) 

109 elif target_type == "number": 

110 return float(original_item) 

111 elif target_type == "integer": 

112 return int(original_item) 

113 elif target_type == "boolean": 

114 if isinstance(original_item, str): 

115 return strtobool(original_item) == 1 

116 return bool(original_item) 

117 except ValueError: 

118 return original_item 

119 return original_item

121 def __get_normalizer(self, schema_key: str, original_validator: Callable): 

122 """ 

123 Traverse through object fields using native jsonschema validator and apply normalization function. 

124 :param schema_key related json schema key that currently being validated/normalized. 

125 :original_validator: native jsonschema validator callback. 

126 """ 

127 

128 def normalizator(validator_instance: Callable, val: Any, instance: Any, schema: Dict[str, Any]): 

129 """ 

130 Jsonschema validator callable it uses for validating instance. We 

131 override default Draft7Validator to perform value transformation 

132 before validation take place. We do not take any action except 

133 logging warn if object does not conform to json schema, just using 

134 jsonschema algorithm to traverse through object fields. 

135 Look 

136 https://python-jsonschema.readthedocs.io/en/stable/creating/?highlight=validators.create#jsonschema.validators.create 

137 validators parameter for detailed description. 

138 : 

139 """ 

140 

141 def resolve(subschema): 

142 if "$ref" in subschema: 

143 _, resolved = validator_instance.resolver.resolve(subschema["$ref"]) 

144 return resolved 

145 return subschema 

146 

147 if schema_key == "type" and instance is not None: 

148 if "object" in val and isinstance(instance, dict): 

149 for k, subschema in schema.get("properties", {}).items(): 

150 if k in instance: 

151 subschema = resolve(subschema) 

152 instance[k] = self.__normalize(instance[k], subschema) 

153 elif "array" in val and isinstance(instance, list): 

154 subschema = schema.get("items", {}) 

155 subschema = resolve(subschema) 

156 for index, item in enumerate(instance): 

157 instance[index] = self.__normalize(item, subschema) 

158 # Running native jsonschema traverse algorithm after field normalization is done. 

159 yield from original_validator(validator_instance, val, instance, schema) 

160 

161 return normalizator

169 if TransformConfig.NoTransform in self._config:
