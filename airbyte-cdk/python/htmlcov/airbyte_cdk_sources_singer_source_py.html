116 def get_sync_mode_overrides(self) -> Dict[str, SyncModeInfo]: 

117 """ 

118 The Singer Spec outlines a way for taps to declare in their catalog that their streams support incremental sync (valid-replication-keys, 

119 forced-replication-method, and others). However, many taps which are incremental don't actually declare that via the catalog, and just 

120 use their input state to perform an incremental sync without giving any hints to the user. An Airbyte Connector built on top of such a 

121 Singer Tap cannot automatically detect which streams are full refresh or incremental or what their cursors are. In those cases the developer 

122 needs to manually specify information about the sync modes. 

123 

124 This method provides a way of doing that: the dict of stream names to SyncModeInfo returned from this method will be used to override each 

125 stream's sync mode information in the Airbyte Catalog output from the discover method. Only set fields provided in the SyncModeInfo are used. 

126 If a SyncModeInfo field is not set, it will not be overridden in the output catalog. 

127 

128 :return: A dict from stream name to the sync modes that should be applied to this stream. 

129 """ 

130 return {}

152 def check_config(self, logger: AirbyteLogger, config_path: str, config: json) -> AirbyteConnectionStatus: 

153 try: 

154 self.try_connect(logger, config)

156 logger.error(f"Exception while connecting to {self.tap_name}: {err}") 

157 # this should be in UI 

158 error_msg = f"Unable to connect to {self.tap_name} with the provided credentials. Error: {err}" 

159 return AirbyteConnectionStatus(status=Status.FAILED, message=error_msg) 

160 return AirbyteConnectionStatus(status=Status.SUCCEEDED)

165 def read_cmd(self, logger: AirbyteLogger, config_path: str, catalog_path: str, state_path: str = None) -> str: 

166 state_path = None if self.force_full_refresh else state_path 

167 args = {"--config": config_path, "--catalog": catalog_path, "--state": state_path} 

168 cmd = " ".join([f"{k} {v}" for k, v in args.items() if v is not None]) 

169 

170 return f"{self.tap_cmd} {cmd}"
