26 def get_schema(self, name: str) -> dict: 

27 """ 

28 This method retrieves a JSON schema from the schemas/ folder. 

29 

30 

31 The expected file structure is to have all top-level schemas (corresponding to streams) in the "schemas/" folder, with any shared $refs 

32 living inside the "schemas/shared/" folder. For example: 

33 

34 schemas/shared/<shared_definition>.json 

35 schemas/<name>.json # contains a $ref to shared_definition 

36 schemas/<name2>.json # contains a $ref to shared_definition 

37 """ 

38 

39 schema_filename = f"schemas/{name}.json" 

40 raw_file = pkgutil.get_data(self.package_name, schema_filename) 

41 if not raw_file: 

42 raise IOError(f"Cannot find file {schema_filename}") 

43 try: 

44 raw_schema = json.loads(raw_file) 

45 except ValueError: 

46 # TODO use proper logging 

47 print(f"Invalid JSON file format for file {schema_filename}") 

48 raise 

49 

50 return self.__resolve_schema_references(raw_schema)

59 class JsonFileLoader: 

60 """ 

61 Custom json file loader to resolve references to resources located in "shared" directory. 

62 We need this for compatability with existing schemas cause all of them have references 

63 pointing to shared_schema.json file instead of shared/shared_schema.json 

64 """ 

65 

66 def __init__(self, uri_base: str, shared: str): 

67 self.shared = shared 

68 self.uri_base = uri_base 

69 

70 def __call__(self, uri: str) -> Dict[str, Any]: 

71 uri = uri.replace(self.uri_base, f"{self.uri_base}/{self.shared}/") 

72 return json.load(open(uri)) 

73 

74 package = importlib.import_module(self.package_name) 

75 base = os.path.dirname(package.__file__) + "/"

77 def create_definitions(obj: dict, definitions: dict) -> Dict[str, Any]: 

78 """ 

79 Scan resolved schema and compose definitions section, also convert 

80 jsonref.JsonRef object to JSON serializable dict. 

81 :param obj - jsonschema object with ref field resovled. 

82 :definitions - object for storing generated definitions. 

83 :return JSON serializable object with references without external dependencies. 

84 """ 

85 if isinstance(obj, jsonref.JsonRef): 

86 def_key = obj.__reference__["$ref"] 

87 def_key = def_key.replace("#/definitions/", "").replace(".json", "_") 

88 definition = create_definitions(obj.__subject__, definitions) 

89 # Omit existance definitions for extenal resource since 

90 # we dont need it anymore. 

91 definition.pop("definitions", None) 

92 definitions[def_key] = definition 

93 return {"$ref": "#/definitions/" + def_key} 

94 elif isinstance(obj, dict): 

95 return {k: create_definitions(v, definitions) for k, v in obj.items()} 

96 elif isinstance(obj, list): 

97 return [create_definitions(item, definitions) for item in obj] 

98 else: 

99 return obj 

100 

101 resolved = jsonref.JsonRef.replace_refs(raw_schema, loader=JsonFileLoader(base, "schemas/shared"), base_uri=base) 

102 definitions = {} 

103 resolved = create_definitions(resolved, definitions) 

104 if definitions: 

105 resolved["definitions"] = definitions 

106 return resolved

109def check_config_against_spec_or_exit(config: Mapping[str, Any], spec: ConnectorSpecification, logger: AirbyteLogger): 

110 """ 

111 Check config object against spec. In case of spec is invalid, throws 

112 an exception with validation error description. 

113 :param config - config loaded from file specified over command line 

114 :param spec - spec object generated by connector 

115 :param logger - Airbyte logger for reporting validation error 

116 """ 

117 spec_schema = spec.connectionSpecification

119 validate(instance=config, schema=spec_schema)

133def split_config(config: Mapping[str, Any]) -> Tuple[dict, InternalConfig]: 

134 """ 

135 Break config map object into 2 instances: first is a dict with user defined 

136 configuration and second is internal config that contains private keys for 

137 acceptance test configuration. 

138 :param 

139 config - Dict object that has been loaded from config file. 

140 :return tuple of user defined config dict with filtered out internal 

141 parameters and SAT internal config object. 

142 """ 

143 main_config = {} 

144 internal_config = {} 

145 for k, v in config.items(): 

146 if k in InternalConfig.KEYWORDS: 

147 internal_config[k] = v 

148 else: 

149 main_config[k] = v 

150 return main_config, InternalConfig.parse_obj(internal_config)
