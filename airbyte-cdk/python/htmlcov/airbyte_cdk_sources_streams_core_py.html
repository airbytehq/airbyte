63 def as_airbyte_stream(self) -> AirbyteStream: 

64 stream = AirbyteStream(name=self.name, json_schema=dict(self.get_json_schema()), supported_sync_modes=[SyncMode.full_refresh]) 

65 

66 if self.supports_incremental: 

67 stream.source_defined_cursor = self.source_defined_cursor 

68 stream.supported_sync_modes.append(SyncMode.incremental) # type: ignore 

69 stream.default_cursor_field = self._wrapped_cursor_field() 

70 

71 keys = Stream._wrapped_primary_key(self.primary_key) 

72 if keys and len(keys) > 0: 

73 stream.source_defined_primary_key = keys 

74 

75 return stream

121 @property 

122 def state_checkpoint_interval(self) -> Optional[int]: 

123 """ 

124 Decides how often to checkpoint state (i.e: emit a STATE message). E.g: if this returns a value of 100, then state is persisted after reading 

125 100 records, then 200, 300, etc.. A good default value is 1000 although your mileage may vary depending on the underlying data source. 

126 

127 Checkpointing a stream avoids re-reading records in the case a sync is failed or cancelled. 

128 

129 return None if state should not be checkpointed e.g: because records returned from the underlying data source are not returned in 

130 ascending order with respect to the cursor field. This can happen if the source does not support reading records in ascending order of 

131 created_at date (or whatever the cursor is). In those cases, state must only be saved once the full stream has been read. 

132 """ 

133 return None

150 @staticmethod 

151 def _wrapped_primary_key(keys: Optional[Union[str, List[str], List[List[str]]]]) -> Optional[List[List[str]]]: 

152 """ 

153 :return: wrap the primary_key property in a list of list of strings required by the Airbyte Stream object. 

154 """ 

155 if not keys: 

156 return None 

157 

158 if isinstance(keys, str): 

159 return [[keys]] 

160 elif isinstance(keys, list): 

161 wrapped_keys = [] 

162 for component in keys: 

163 if isinstance(component, str): 

164 wrapped_keys.append([component]) 

165 elif isinstance(component, list): 

166 wrapped_keys.append(component) 

167 else: 

168 raise ValueError("Element must be either list or str.") 

169 return wrapped_keys 

170 else: 

171 raise ValueError("Element must be either list or str.")
