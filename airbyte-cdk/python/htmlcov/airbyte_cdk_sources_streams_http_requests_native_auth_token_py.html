11class MultipleTokenAuthenticator(AuthBase): 

12 """ 

13 Builds auth header, based on the list of tokens provided. 

14 Auth header is changed per each `get_auth_header` call, using each token in cycle. 

15 The token is attached to each request via the `auth_header` header. 

16 """ 

17 

18 def __init__(self, tokens: List[str], auth_method: str = "Bearer", auth_header: str = "Authorization"): 

19 self.auth_method = auth_method 

20 self.auth_header = auth_header 

21 self._tokens = tokens 

22 self._tokens_iter = cycle(self._tokens) 

23 

24 def __call__(self, request): 

25 request.headers.update(self.get_auth_header()) 

26 return request 

27 

28 def get_auth_header(self) -> Mapping[str, Any]: 

29 return {self.auth_header: f"{self.auth_method} {next(self._tokens_iter)}"}
