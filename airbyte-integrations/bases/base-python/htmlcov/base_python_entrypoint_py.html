20class AirbyteEntrypoint(object): 

21 def __init__(self, source: Source): 

22 self.source = source 

23 

24 def start(self, args): 

25 # set up parent parsers 

26 parent_parser = argparse.ArgumentParser(add_help=False) 

27 main_parser = argparse.ArgumentParser() 

28 subparsers = main_parser.add_subparsers(title="commands", dest="command") 

29 

30 # spec 

31 subparsers.add_parser("spec", help="outputs the json configuration specification", parents=[parent_parser]) 

32 

33 # check 

34 check_parser = subparsers.add_parser("check", help="checks the config can be used to connect", parents=[parent_parser]) 

35 required_check_parser = check_parser.add_argument_group("required named arguments") 

36 required_check_parser.add_argument("--config", type=str, required=True, help="path to the json configuration file") 

37 

38 # discover 

39 discover_parser = subparsers.add_parser( 

40 "discover", help="outputs a catalog describing the source's schema", parents=[parent_parser] 

41 ) 

42 required_discover_parser = discover_parser.add_argument_group("required named arguments") 

43 required_discover_parser.add_argument("--config", type=str, required=True, help="path to the json configuration file") 

44 

45 # read 

46 read_parser = subparsers.add_parser("read", help="reads the source and outputs messages to STDOUT", parents=[parent_parser]) 

47 

48 read_parser.add_argument("--state", type=str, required=False, help="path to the json-encoded state file") 

49 required_read_parser = read_parser.add_argument_group("required named arguments") 

50 required_read_parser.add_argument("--config", type=str, required=True, help="path to the json configuration file") 

51 required_read_parser.add_argument( 

52 "--catalog", type=str, required=True, help="path to the catalog used to determine which data to read" 

53 ) 

54 

55 # parse the args 

56 parsed_args = main_parser.parse_args(args) 

57 

58 # execute 

59 cmd = parsed_args.command 

60 if not cmd: 

61 raise Exception("No command passed") 

62 

63 # todo: add try catch for exceptions with different exit codes 

64 

65 with tempfile.TemporaryDirectory() as temp_dir: 

66 if cmd == "spec": 

67 message = AirbyteMessage(type=Type.SPEC, spec=self.source.spec(logger)) 

68 print(message.json(exclude_unset=True)) 

69 sys.exit(0) 

70 

71 raw_config = self.source.read_config(parsed_args.config) 

72 config = self.source.configure(raw_config, temp_dir) 

73 

74 if cmd == "check": 

75 check_result = self.source.check(logger, config) 

76 if check_result.status == Status.SUCCEEDED: 

77 logger.info("Check succeeded") 

78 else: 

79 logger.error("Check failed") 

80 

81 output_message = AirbyteMessage(type=Type.CONNECTION_STATUS, connectionStatus=check_result).json(exclude_unset=True) 

82 print(output_message) 

83 sys.exit(0) 

84 elif cmd == "discover": 

85 catalog = self.source.discover(logger, config) 

86 print(AirbyteMessage(type=Type.CATALOG, catalog=catalog).json(exclude_unset=True)) 

87 sys.exit(0) 

88 elif cmd == "read": 

89 catalog = self.source.read_catalog(parsed_args.catalog) 

90 state = self.source.read_state(parsed_args.state) 

91 generator = self.source.read(logger, config, catalog, state) 

92 for message in generator: 

93 print(message.json(exclude_unset=True)) 

94 sys.exit(0) 

95 else: 

96 raise Exception("Unexpected command " + cmd)

104def main(): 

105 impl_module = os.environ.get("AIRBYTE_IMPL_MODULE", Source.__module__) 

106 impl_class = os.environ.get("AIRBYTE_IMPL_PATH", Source.__name__) 

107 module = importlib.import_module(impl_module) 

108 impl = getattr(module, impl_class) 

109 

110 # set up and run entrypoint 

111 source = impl() 

112 

113 if not isinstance(source, Source): 

114 raise Exception("Source implementation provided does not implement Source class!") 

115 

116 launch(source, sys.argv[1:])
