26from airbyte_protocol import AirbyteCatalog, SyncMode 

27 

28 

29class CatalogHelper: 

30 @staticmethod 

31 def coerce_catalog_as_full_refresh(catalog: AirbyteCatalog) -> AirbyteCatalog: 

32 """ 

33 Updates the sync mode on all streams in this catalog to be full refresh 

34 """ 

35 coerced_catalog = catalog.copy() 

36 for stream in catalog.streams: 

37 stream.source_defined_cursor = False 

38 stream.supported_sync_modes = [SyncMode.full_refresh] 

39 stream.default_cursor_field = None 

40 

41 # remove nulls 

42 return AirbyteCatalog.parse_raw(coerced_catalog.json(exclude_unset=True, exclude_none=True))
