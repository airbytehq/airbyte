62 self._schema_loader = self.schema_loader_class(package_name) 

63 self._stream_methods = self._enumerate_methods()

65 def _enumerate_methods(self) -> Mapping[str, callable]: 

66 """Detect available streams and return mapping""" 

67 prefix = "stream__" 

68 mapping = {} 

69 methods = inspect.getmembers(self.__class__, predicate=inspect.isfunction) 

70 for name, method in methods: 

71 if name.startswith(prefix): 

72 mapping[name[len(prefix) :]] = getattr(self, name) 

73 

74 return mapping

76 @staticmethod 

77 def _get_fields_from_stream(stream: AirbyteStream) -> List[str]: 

78 return list(stream.json_schema.get("properties", {}).keys())

80 def _get_stream_method(self, name: str) -> Callable: 

81 method = self._stream_methods.get(name) 

82 if not method: 

83 raise ValueError(f"Client does not know how to read stream `{name}`") 

84 return method

86 def read_stream(self, stream: AirbyteStream) -> Generator[Dict[str, Any], None, None]: 

87 """Yield records from stream""" 

88 method = self._get_stream_method(stream.name) 

89 fields = self._get_fields_from_stream(stream) 

90 

91 for message in method(fields=fields): 

92 yield dict(message)

94 @property 

95 def streams(self) -> Generator[AirbyteStream, None, None]: 

96 """List of available streams""" 

97 for name, method in self._stream_methods.items(): 

98 supported_sync_modes = [SyncMode.full_refresh] 

99 source_defined_cursor = False 

100 if self.stream_has_state(name): 

101 supported_sync_modes += [SyncMode.incremental] 

102 source_defined_cursor = True 

103 

104 yield AirbyteStream( 

105 name=name, 

106 json_schema=self._schema_loader.get_schema(name), 

107 supported_sync_modes=supported_sync_modes,
