28class AbstractSource(Source, ABC): 

29 """ 

30 Abstract base class for an Airbyte Source. Consumers should implement any abstract methods 

31 in this class to create an Airbyte Specification compliant Source. 

32 """ 

33 

34 @abstractmethod 

35 def check_connection(self, logger: AirbyteLogger, config: Mapping[str, Any]) -> Tuple[bool, Optional[Any]]: 

36 """ 

37 :param config: The user-provided configuration as specified by the source's spec. This usually contains information required to check connection e.g. tokens, secrets and keys etc. 

38 :return: A tuple of (boolean, error). If boolean is true, then the connection check is successful and we can connect to the underlying data 

39 source using the provided configuration. 

40 Otherwise, the input config cannot be used to connect to the underlying data source, and the "error" object should describe what went wrong. 

41 The error object will be cast to string to display the problem to the user. 

42 """ 

43 

44 @abstractmethod 

45 def streams(self, config: Mapping[str, Any]) -> List[Stream]: 

46 """ 

47 :param config: The user-provided configuration as specified by the source's spec. Any stream construction related operation should happen here. 

48 :return: A list of the streams in this source connector. 

49 """

51 @property 

52 def name(self) -> str: 

53 """Source name""" 

54 return self.__class__.__name__

56 def discover(self, logger: AirbyteLogger, config: Mapping[str, Any]) -> AirbyteCatalog: 

57 """Implements the Discover operation from the Airbyte Specification. See https://docs.airbyte.io/architecture/airbyte-specification.""" 

58 streams = [stream.as_airbyte_stream() for stream in self.streams(config=config)] 

59 return AirbyteCatalog(streams=streams)

61 def check(self, logger: AirbyteLogger, config: Mapping[str, Any]) -> AirbyteConnectionStatus: 

62 """Implements the Check Connection operation from the Airbyte Specification. See https://docs.airbyte.io/architecture/airbyte-specification.""" 

63 try: 

64 check_succeeded, error = self.check_connection(logger, config) 

65 if not check_succeeded: 

66 return AirbyteConnectionStatus(status=Status.FAILED, message=str(error)) 

67 except Exception as e: 

68 return AirbyteConnectionStatus(status=Status.FAILED, message=str(e)) 

69 

70 return AirbyteConnectionStatus(status=Status.SUCCEEDED)

72 def read( 

73 self, logger: AirbyteLogger, config: Mapping[str, Any], catalog: ConfiguredAirbyteCatalog, state: MutableMapping[str, Any] = None 

74 ) -> Iterator[AirbyteMessage]: 

75 """Implements the Read operation from the Airbyte Specification. See https://docs.airbyte.io/architecture/airbyte-specification.""" 

76 connector_state = copy.deepcopy(state or {}) 

77 logger.info(f"Starting syncing {self.name}") 

78 # TODO assert all streams exist in the connector 

79 # get the streams once in case the connector needs to make any queries to generate them 

80 stream_instances = {s.name: s for s in self.streams(config)} 

81 for configured_stream in catalog.streams: 

82 try: 

83 stream_instance = stream_instances[configured_stream.stream.name] 

84 yield from self._read_stream( 

85 logger=logger, stream_instance=stream_instance, configured_stream=configured_stream, connector_state=connector_state 

86 ) 

87 except Exception as e: 

88 logger.exception(f"Encountered an exception while reading stream {self.name}") 

89 raise e 

90 

91 logger.info(f"Finished syncing {self.name}")

93 def _read_stream( 

94 self, 

95 logger: AirbyteLogger, 

96 stream_instance: Stream, 

97 configured_stream: ConfiguredAirbyteStream, 

98 connector_state: MutableMapping[str, Any], 

99 ) -> Iterator[AirbyteMessage]:

102 if use_incremental: 

103 record_iterator = self._read_incremental(logger, stream_instance, configured_stream, connector_state)

105 record_iterator = self._read_full_refresh(stream_instance, configured_stream)

107 record_counter = 0 

108 stream_name = configured_stream.stream.name 

109 logger.info(f"Syncing stream: {stream_name} ") 

110 for record in record_iterator: 

111 if record.type == MessageType.RECORD: 

112 record_counter += 1 

113 yield record

115 logger.info(f"Read {record_counter} records from {stream_name} stream")

150 def _read_full_refresh(self, stream_instance: Stream, configured_stream: ConfiguredAirbyteStream) -> Iterator[AirbyteMessage]: 

151 args = {"sync_mode": SyncMode.full_refresh, "cursor_field": configured_stream.cursor_field} 

152 for slices in stream_instance.stream_slices(**args): 

153 for record in stream_instance.read_records(stream_slice=slices, **args): 

154 yield self._as_airbyte_record(configured_stream.stream.name, record)

156 def _checkpoint_state(self, stream_name, stream_state, connector_state, logger): 

157 logger.info(f"Setting state of {stream_name} stream to {stream_state}") 

158 connector_state[stream_name] = stream_state 

159 return AirbyteMessage(type=MessageType.STATE, state=AirbyteStateMessage(data=connector_state))

161 def _as_airbyte_record(self, stream_name: str, data: Mapping[str, Any]): 

162 now_millis = int(datetime.now().timestamp()) * 1000 

163 message = AirbyteRecordMessage(stream=stream_name, data=data, emitted_at=now_millis) 

164 return AirbyteMessage(type=MessageType.RECORD, record=message)
