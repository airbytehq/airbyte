27 def run(self, args): 

28 inputs = self.parse(args) 

29 original_config = self.read_json_config(inputs["config"]) 

30 integration_type = inputs["integration_type"] 

31 

32 transformed_dbt_project = self.transform_dbt_project(integration_type) 

33 self.write_yaml_config(inputs["output_path"], transformed_dbt_project, "dbt_project.yml") 

34 

35 transformed_config = self.transform(integration_type, original_config) 

36 self.write_yaml_config(inputs["output_path"], transformed_config, "profiles.yml") 

37 if self.is_ssh_tunnelling(original_config): 

38 self.write_ssh_config(inputs["output_path"], original_config, transformed_config)

58 def transform_dbt_project(self, integration_type: DestinationType): 

59 data = pkgutil.get_data(self.__class__.__module__.split(".")[0], "transform_config/dbt_project_base.yml") 

60 if not data: 

61 raise FileExistsError("Failed to load profile_base.yml") 

62 base_project = yaml.load(data, Loader=yaml.FullLoader) 

63 

64 if integration_type.value == DestinationType.oracle.value: 

65 base_project["quoting"]["database"] = False 

66 base_project["quoting"]["identifier"] = False 

67 

68 return base_project

70 def transform(self, integration_type: DestinationType, config: Dict[str, Any]): 

71 data = pkgutil.get_data(self.__class__.__module__.split(".")[0], "transform_config/profile_base.yml") 

72 if not data: 

73 raise FileExistsError("Failed to load profile_base.yml") 

74 base_profile = yaml.load(data, Loader=yaml.FullLoader) 

75 

76 transformed_integration_config = { 

77 DestinationType.bigquery.value: self.transform_bigquery, 

78 DestinationType.postgres.value: self.transform_postgres, 

79 DestinationType.redshift.value: self.transform_redshift, 

80 DestinationType.snowflake.value: self.transform_snowflake, 

81 DestinationType.mysql.value: self.transform_mysql, 

82 DestinationType.oracle.value: self.transform_oracle, 

83 }[integration_type.value](config) 

84 

85 # merge pre-populated base_profile with destination-specific configuration. 

86 base_profile["normalize"]["outputs"]["prod"] = transformed_integration_config 

87 

88 return base_profile

91 def is_ssh_tunnelling(config: Dict[str, Any]) -> bool: 

92 tunnel_methods = ["SSH_KEY_AUTH", "SSH_PASSWORD_AUTH"] 

93 if ( 

94 "tunnel_method" in config.keys() 

95 and "tunnel_method" in config["tunnel_method"] 

96 and config["tunnel_method"]["tunnel_method"].upper() in tunnel_methods 

97 ): 

98 return True 

99 else: 

100 return False

102 @staticmethod 

103 def is_port_free(port: int) -> bool: 

104 with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s: 

105 try: 

106 s.bind(("localhost", port)) 

107 except Exception as e: 

108 print(f"port {port} unsuitable: {e}") 

109 return False 

110 else: 

111 print(f"port {port} is free") 

112 return True

114 @staticmethod 

115 def pick_a_port() -> int: 

116 """ 

117 This function finds a free port, starting with 50001 and adding 1 until we find an open port. 

118 """ 

119 port_to_check = 50001 # just past start of dynamic port range (49152:65535) 

120 while not TransformConfig.is_port_free(port_to_check): 

121 port_to_check += 1 

122 # error if we somehow hit end of port range 

123 if port_to_check > 65535: 

124 raise RuntimeError("Couldn't find a free port to use.") 

125 return port_to_check

127 @staticmethod 

128 def get_ssh_altered_config(config: Dict[str, Any], port_key: str = "port", host_key: str = "host") -> Dict[str, Any]: 

129 """ 

130 This should be called only if ssh tunneling is on. 

131 It will return config with appropriately altered port and host values 

132 """ 

133 # make a copy of config rather than mutate in place 

134 ssh_ready_config = {k: v for k, v in config.items()} 

135 ssh_ready_config[port_key] = TransformConfig.pick_a_port() 

136 ssh_ready_config[host_key] = "localhost" 

137 return ssh_ready_config
