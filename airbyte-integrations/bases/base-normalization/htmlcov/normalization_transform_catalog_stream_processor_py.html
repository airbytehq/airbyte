56class StreamProcessor(object): 

57 """ 

58 Takes as input an Airbyte Stream as described in the (configured) Airbyte Catalog's Json Schema. 

59 Associated input raw data is expected to be stored in a staging area table. 

60 

61 This processor generates SQL models to transform such a stream into a final table in the destination schema. 

62 This is done by generating a DBT pipeline of transformations (multiple SQL models queries) that may be materialized 

63 in the intermediate schema "raw_schema" (changing the dbt_project.yml settings). 

64 The final output data should be written in "schema". 

65 

66 The pipeline includes transformations such as: 

67 - Parsing a JSON blob column and extracting each field property in its own SQL column 

68 - Casting each SQL column to the proper JSON data type 

69 - Generating an artificial (primary key) ID column based on the hashing of the row 

70 

71 If any nested columns are discovered in the stream, a JSON blob SQL column is created in the top level parent stream 

72 and a new StreamProcessor instance will be spawned for each children substreams. These Sub-Stream Processors are then 

73 able to generate models to parse and extract recursively from its parent StreamProcessor model into separate SQL tables 

74 the content of that JSON blob SQL column. 

75 """

211 def process(self) -> List["StreamProcessor"]: 

212 """ 

213 See description of StreamProcessor class. 

214 @return List of StreamProcessor to handle recursively nested columns from this stream 

215 """ 

216 # Check properties 

217 if not self.properties: 

218 print(f" Ignoring stream '{self.stream_name}' from {self.current_json_path()} because properties list is empty") 

219 return [] 

220 

221 column_names = self.extract_column_names() 

222 column_count = len(column_names) 

223 

224 if column_count == 0: 

225 print(f" Ignoring stream '{self.stream_name}' from {self.current_json_path()} because no columns were identified") 

226 return [] 

227 

228 from_table = self.from_table 

229 # Transformation Pipeline for this stream 

230 from_table = self.add_to_outputs(self.generate_json_parsing_model(from_table, column_names), is_intermediate=True, suffix="ab1") 

231 from_table = self.add_to_outputs( 

232 self.generate_column_typing_model(from_table, column_names), is_intermediate=True, column_count=column_count, suffix="ab2" 

233 ) 

234 from_table = self.add_to_outputs( 

235 self.generate_id_hashing_model(from_table, column_names), is_intermediate=True, column_count=column_count, suffix="ab3" 

236 ) 

237 if self.destination_sync_mode.value == DestinationSyncMode.append_dedup.value: 

238 from_table = self.add_to_outputs(self.generate_dedup_record_model(from_table, column_names), is_intermediate=True, suffix="ab4") 

239 if self.destination_type == DestinationType.ORACLE: 

240 where_clause = '\nwhere "_AIRBYTE_ROW_NUM" = 1' 

241 else: 

242 where_clause = "\nwhere _airbyte_row_num = 1" 

243 from_table = self.add_to_outputs( 

244 self.generate_scd_type_2_model(from_table, column_names) + where_clause, 

245 is_intermediate=False, 

246 column_count=column_count, 

247 suffix="scd", 

248 ) 

249 if self.destination_type == DestinationType.ORACLE: 

250 where_clause = '\nwhere "_AIRBYTE_ACTIVE_ROW" = 1' 

251 else: 

252 where_clause = "\nwhere _airbyte_active_row = 1" 

253 

254 from_table = self.add_to_outputs( 

255 self.generate_final_model(from_table, column_names) + where_clause, is_intermediate=False, column_count=column_count 

256 ) 

257 # TODO generate yaml file to dbt test final table where primary keys should be unique 

258 else: 

259 from_table = self.add_to_outputs( 

260 self.generate_final_model(from_table, column_names), is_intermediate=False, column_count=column_count 

261 ) 

262 return self.find_children_streams(from_table, column_names)

264 def extract_column_names(self) -> Dict[str, Tuple[str, str]]: 

265 """ 

266 Generate a mapping of JSON properties to normalized SQL Column names, handling collisions and avoid duplicate names 

267 

268 The mapped value to a field property is a tuple where: 

269 - the first value is the normalized "raw" column name 

270 - the second value is the normalized quoted column name to be used in jinja context 

271 """ 

272 fields = [] 

273 for field in self.properties.keys(): 

274 if not is_airbyte_column(field): 

275 fields.append(field) 

276 result = {} 

277 field_names = set() 

278 for field in fields: 

279 field_name = self.name_transformer.normalize_column_name(field, in_jinja=False) 

280 jinja_name = self.name_transformer.normalize_column_name(field, in_jinja=True) 

281 if field_name in field_names: 

282 # TODO handle column name duplicates or collisions deterministically in this stream 

283 for i in range(1, 1000): 

284 field_name = self.name_transformer.normalize_column_name(f"{field}_{i}", in_jinja=False) 

285 jinja_name = self.name_transformer.normalize_column_name(f"{field}_{i}", in_jinja=True) 

286 if field_name not in field_names: 

287 break 

288 field_names.add(field_name) 

289 result[field] = (field_name, jinja_name) 

290 return result

295 The current stream/table is used as the parent from which to extract data from.

297 properties = self.properties

299 for field in properties.keys(): 

300 children_properties = None 

301 if is_airbyte_column(field): 

302 pass 

303 elif is_combining_node(properties[field]): 

304 # TODO: merge properties of all combinations 

305 pass 

306 elif "type" not in properties[field] or is_object(properties[field]["type"]): 

307 # properties without 'type' field are treated like properties with 'type' = 'object' 

308 children_properties = find_properties_object([], field, properties[field]) 

309 is_nested_array = False 

310 json_column_name = column_names[field][1] 

311 elif is_array(properties[field]["type"]) and "items" in properties[field]: 

312 quoted_field = column_names[field][1] 

313 children_properties = find_properties_object([], field, properties[field]["items"]) 

314 is_nested_array = True 

315 json_column_name = f"unnested_column_value({quoted_field})" 

316 if children_properties: 

317 for child_key in children_properties: 

318 stream_processor = StreamProcessor.create_from_parent( 

319 parent=self, 

320 child_name=field, 

321 json_column_name=json_column_name, 

322 properties=children_properties[child_key], 

323 is_nested_array=is_nested_array, 

324 from_table=from_table, 

325 ) 

326 children.append(stream_processor) 

327 return children

420 def cast_property_type(self, property_name: str, column_name: str, jinja_column: str) -> str: 

421 definition = self.properties[property_name] 

422 if "type" not in definition: 

423 print(f"WARN: Unknown type for column {property_name} at {self.current_json_path()}") 

424 return column_name 

425 elif is_array(definition["type"]): 

426 return column_name 

427 elif is_object(definition["type"]): 

428 sql_type = jinja_call("type_json()") 

429 # Treat simple types from narrower to wider scope type: boolean < integer < number < string 

430 elif is_boolean(definition["type"]): 

431 cast_operation = jinja_call(f"cast_to_boolean({jinja_column})") 

432 return f"{cast_operation} as {column_name}" 

433 elif is_integer(definition["type"]): 

434 sql_type = jinja_call("dbt_utils.type_bigint()") 

435 elif is_number(definition["type"]): 

436 sql_type = jinja_call("dbt_utils.type_float()") 

437 elif is_timestamp_with_time_zone(definition): 

438 sql_type = jinja_call("type_timestamp_with_timezone()") 

439 elif is_date(definition): 

440 sql_type = jinja_call("type_date()") 

441 elif is_string(definition["type"]): 

442 sql_type = jinja_call("dbt_utils.type_string()") 

443 else: 

444 print(f"WARN: Unknown type {definition['type']} for column {property_name} at {self.current_json_path()}") 

445 return column_name 

446 

447 return f"cast({column_name} as {sql_type}) as {column_name}"

486 def safe_cast_to_string(definition: Dict, column_name: str, destination_type: DestinationType) -> str: 

487 """ 

488 Note that the result from this static method should always be used within a 

489 jinja context (for example, from jinja macro surrogate_key call) 

490 

491 The jinja_remove function is necessary because of Oracle database, some columns 

492 are created with {{ quote('column_name') }} and reused the same fields for this 

493 operation. Because the quote is injected inside a jinja macro we need to remove 

494 the curly brackets. 

495 """ 

496 

497 if "type" not in definition: 

498 col = column_name 

499 elif is_boolean(definition["type"]): 

500 col = f"boolean_to_string({column_name})" 

501 elif is_array(definition["type"]): 

502 col = f"array_to_string({column_name})" 

503 else: 

504 col = column_name 

505 

506 if destination_type == DestinationType.ORACLE: 

507 quote_in_parenthesis = re.compile(r"quote\((.*)\)") 

508 return remove_jinja(col) if quote_in_parenthesis.findall(col) else col 

509 

510 return col

538 def generate_scd_type_2_model(self, from_table: str, column_names: Dict[str, Tuple[str, str]]) -> str: 

539 

540 scd_sql_template = """ 

541-- SQL model to build a Type 2 Slowly Changing Dimension (SCD) table for each record identified by their primary key 

542select 

543 {%- if parent_hash_id %} 

544 {{ parent_hash_id }}, 

545 {%- endif %} 

546 {%- for field in fields %} 

547 {{ field }}, 

548 {%- endfor %} 

549 {{ cursor_field }} as {{ airbyte_start_at }}, 

550 lag({{ cursor_field }}) over ( 

551 partition by {{ primary_key }} 

552 order by {{ cursor_field }} {{ order_null }}, {{ cursor_field }} desc, {{ col_emitted_at }} desc 

553 ) as {{ airbyte_end_at }}, 

554 case when lag({{ cursor_field }}) over ( 

555 partition by {{ primary_key }} 

556 order by {{ cursor_field }} {{ order_null }}, {{ cursor_field }} desc, {{ col_emitted_at }} desc{{ cdc_updated_at_order }} 

557 ) is null {{ cdc_active_row }} then 1 else 0 end as {{ active_row }}, 

558 {{ col_emitted_at }}, 

559 {{ hash_id }} 

560from {{ from_table }} 

561{{ sql_table_comment }} 

562 """ 

563 

564 template = Template(scd_sql_template) 

565 

566 order_null = "is null asc" 

567 if self.destination_type == DestinationType.ORACLE: 

568 order_null = "asc nulls first" 

569 

570 cdc_active_row_pattern = "" 

571 cdc_updated_order_pattern = "" 

572 if "_ab_cdc_deleted_at" in column_names.keys(): 

573 col_cdc_deleted_at = self.name_transformer.normalize_column_name("_ab_cdc_deleted_at") 

574 col_cdc_updated_at = self.name_transformer.normalize_column_name("_ab_cdc_updated_at")

598 if not self.cursor_field: 

599 cursor = self.name_transformer.normalize_column_name("_airbyte_emitted_at", in_jinja)

617 def get_primary_key_from_path(self, column_names: Dict[str, Tuple[str, str]], path: List[str]) -> str: 

618 if path and len(path) == 1: 

619 field = path[0] 

620 if not is_airbyte_column(field): 

621 if "type" in self.properties[field]: 

622 property_type = self.properties[field]["type"] 

623 else: 

624 property_type = "object" 

625 if is_number(property_type) or is_object(property_type): 

626 # some destinations don't handle float columns (or complex types) as primary keys, turn them to string 

627 return f"cast({column_names[field][0]} as {jinja_call('dbt_utils.type_string()')})" 

628 else: 

629 return column_names[field][0] 

630 else: 

631 # using an airbyte generated column 

632 return f"cast({field} as {jinja_call('dbt_utils.type_string()')})" 

633 else: 

634 if path: 

635 raise ValueError(f"Unsupported nested path {'.'.join(path)} for stream {self.stream_name}") 

636 else: 

637 raise ValueError(f"No path specified for stream {self.stream_name}")

669 def add_to_outputs(self, sql: str, is_intermediate: bool, column_count: int = 0, suffix: str = "") -> str: 

670 schema = self.get_schema(is_intermediate) 

671 # MySQL table names need to be manually truncated, because it does not do it automatically 

672 truncate_name = self.destination_type == DestinationType.MYSQL 

673 table_name = self.tables_registry.get_table_name(schema, self.json_path, self.stream_name, suffix, truncate_name) 

674 file_name = self.tables_registry.get_file_name(schema, self.json_path, self.stream_name, suffix, truncate_name) 

675 file = f"{file_name}.sql" 

676 if is_intermediate: 

677 if column_count <= MAXIMUM_COLUMNS_TO_USE_EPHEMERAL: 

678 output = os.path.join("airbyte_ctes", self.schema, file) 

679 else: 

680 # dbt throws "maximum recursion depth exceeded" exception at runtime 

681 # if ephemeral is used with large number of columns, use views instead 

682 output = os.path.join("airbyte_views", self.schema, file) 

683 else: 

684 output = os.path.join("airbyte_tables", self.schema, file) 

685 tags = self.get_model_tags(is_intermediate) 

686 # The alias() macro configs a model's final table name. 

687 if file_name != table_name: 

688 header = jinja_call(f'config(alias="{table_name}", schema="{schema}", tags=[{tags}])') 

689 else: 

690 if self.destination_type == DestinationType.ORACLE: 

691 header = jinja_call(f'config(schema="{self.default_schema}", tags=[{tags}])') 

692 else: 

693 header = jinja_call(f'config(schema="{schema}", tags=[{tags}])') 

694 self.sql_outputs[ 

695 output 

696 ] = f""" 

697{header} 

698{sql} 

699""" 

700 json_path = self.current_json_path() 

701 print(f" Generating {output} from {json_path}") 

702 return ref_table(file_name)

787def find_properties_object(path: List[str], field: str, properties) -> Dict[str, Dict]: 

788 """ 

789 This function is trying to look for a nested "properties" node under the current JSON node to 

790 identify all nested objects. 

791 

792 @param path JSON path traversed so far to arrive to this node 

793 @param field is the current field being considered in the Json Tree 

794 @param properties is the child tree of properties of the current field being searched 

795 """ 

796 result = {} 

797 current_path = path + [field] 

798 current = "_".join(current_path) 

799 if isinstance(properties, str) or isinstance(properties, int): 

800 return {} 

801 else: 

802 if "items" in properties: 

803 return find_properties_object(path, field, properties["items"]) 

804 elif "properties" in properties: 

805 # we found a properties object 

806 return {current: properties["properties"]} 

807 elif "type" in properties and is_simple_property(properties["type"]): 

808 # we found a basic type 

809 return {current: {}} 

810 elif isinstance(properties, dict): 

811 for key in properties.keys(): 

812 child = find_properties_object(path=current_path, field=key, properties=properties[key]) 

813 if child: 

814 result.update(child) 

815 elif isinstance(properties, list): 

816 for item in properties: 

817 child = find_properties_object(path=current_path, field=field, properties=item) 

818 if child: 

819 result.update(child) 

820 return result
