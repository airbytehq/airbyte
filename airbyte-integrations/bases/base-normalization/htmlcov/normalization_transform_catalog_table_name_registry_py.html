125class TableNameRegistry: 

126 """ 

127 A registry object that records table names being used during the run 

128 

129 This registry helps detecting naming conflicts/collisions and how to resolve them. 

130 

131 First, we collect all schema/stream_name/json_path listed in the catalog to detect any collisions, whether it is from: 

132 - table naming: truncated stream name could conflict with each other within the same destination schema 

133 - file naming: dbt use a global registry of file names without considering schema, so two tables with the same name in different schema 

134 is valid but dbt would fail to distinguish them. Thus, the file needs should be unique within a dbt project (for example, 

135 by adding the schema name to the file name when such collision occurs?) 

136 

137 To do so, we build list of "simple" names without dealing with any collisions. 

138 Next, we check if/when we encounter such naming conflicts. They usually happen when destination require a certain naming convention 

139 with a limited number of characters, thus, we have to end up truncating names and creating collisions. 

140 

141 In those cases, we resolve collisions using a more complex naming scheme using a suffix generated from hash of full names to make 

142 them short and unique (but hard to remember/use). 

143 """ 

144 

145 def __init__(self, destination_type: DestinationType): 

146 """ 

147 @param destination_type is the destination type of warehouse 

148 """ 

149 self.destination_type: DestinationType = destination_type 

150 self.name_transformer: DestinationNameTransformer = DestinationNameTransformer(destination_type) 

151 # Simple XXX registry are collecting "simple" XXX names (with potential collisions) 

152 self.simple_file_registry: NormalizedFilesRegistry = NormalizedFilesRegistry() 

153 self.simple_table_registry: NormalizedTablesRegistry = NormalizedTablesRegistry(self.name_transformer) 

154 # Registry is the collision free (resolved) mapping of schema json_path of the stream to the names that should be used 

155 self.registry: Dict[str, ResolvedNameMetadata] = {} 

156 

157 def register_table(self, intermediate_schema: str, schema: str, stream_name: str, json_path: List[str]): 

158 """ 

159 Record usages of simple table and file names used by each stream (top level and nested) in both 

160 intermediate_schema and schema. 

161 

162 After going through all streams and sub-streams, we'll be able to find if any collisions are present within 

163 this catalog. 

164 """ 

165 intermediate_schema = self.name_transformer.normalize_schema_name(intermediate_schema, False, False) 

166 schema = self.name_transformer.normalize_schema_name(schema, False, False) 

167 table_name = self.get_simple_table_name(json_path) 

168 self.simple_table_registry.add(intermediate_schema, schema, json_path, stream_name, table_name)

170 def get_simple_table_name(self, json_path: List[str]) -> str: 

171 """ 

172 Generates a simple table name, possibly in collisions within this catalog because of truncation 

173 """ 

174 return self.name_transformer.normalize_table_name("_".join(json_path))

177 conflicts = self.resolve_table_names() 

178 self.resolve_file_names() 

179 return conflicts

181 def resolve_table_names(self) -> List[ConflictedNameMetadata]: 

182 """ 

183 Build a collision free registry from all schema/stream_name/json_path collected so far. 

184 """ 

185 resolved_keys = [] 

186 # deal with table name collisions within the same schema first. 

187 # file name should be equal to table name here 

188 table_count = 0 

189 

190 for key in self.simple_table_registry: 

191 for value in self.simple_table_registry[key]: 

192 table_count += 1 

193 if self.simple_table_registry.has_collisions(key): 

194 # handle collisions with unique hashed names 

195 table_name = self.get_hashed_table_name(value.schema, value.json_path, value.stream_name, value.table_name) 

196 resolved_keys.append(ConflictedNameMetadata(value.schema, value.json_path, value.table_name, table_name)) 

197 else: 

198 table_name = value.table_name 

199 self.registry[self.get_registry_key(value.intermediate_schema, value.json_path, value.stream_name)] = ResolvedNameMetadata( 

200 value.intermediate_schema, 

201 table_name, 

202 # use table_name as file_name for now 

203 table_name, 

204 ) 

205 self.registry[self.get_registry_key(value.schema, value.json_path, value.stream_name)] = ResolvedNameMetadata( 

206 value.schema, 

207 table_name, 

208 # use table_name as file_name for now 

209 table_name, 

210 ) 

211 self.simple_file_registry.add(value.intermediate_schema, value.schema, value.json_path, value.stream_name, table_name) 

212 registry_size = len(self.registry) 

213 

214 # Oracle doesnt support namespace and this break this logic. 

215 if self.destination_type != DestinationType.ORACLE: 

216 assert (table_count * 2) == registry_size, f"Mismatched number of tables {table_count * 2} vs {registry_size} being resolved" 

217 return resolved_keys 

218 

219 def resolve_file_names(self): 

220 # deal with file name collisions across schemas and update the file name to use in the registry when necessary 

221 file_count = 0 

222 for key in self.simple_file_registry: 

223 for value in self.simple_file_registry[key]: 

224 file_count += 1 

225 if self.simple_file_registry.has_collisions(key): 

226 # handle collisions with unique hashed names including schema 

227 self.registry[ 

228 self.get_registry_key(value.intermediate_schema, value.json_path, value.stream_name) 

229 ] = ResolvedNameMetadata( 

230 value.intermediate_schema, value.table_name, self.resolve_file_name(value.intermediate_schema, value.table_name) 

231 ) 

232 self.registry[self.get_registry_key(value.schema, value.json_path, value.stream_name)] = ResolvedNameMetadata( 

233 value.schema, value.table_name, self.resolve_file_name(value.schema, value.table_name) 

234 ) 

235 registry_size = len(self.registry) 

236 

237 # Oracle doesnt support namespace and this break this logic. 

238 if self.destination_type != DestinationType.ORACLE: 

239 assert (file_count * 2) == registry_size, f"Mismatched number of tables {file_count * 2} vs {registry_size} being resolved"

241 def get_hashed_table_name(self, schema: str, json_path: List[str], stream_name: str, table_name: str) -> str: 

242 """ 

243 Generates a unique table name to avoid collisions within this catalog. 

244 This is using a hash of full names but it is hard to use and remember, so this should be done rarely... 

245 We'd prefer to use "simple" names instead as much as possible. 

246 """ 

247 if len(json_path) == 1: 

248 # collisions on a top level stream name, add a hash of schema + stream name to the (truncated?) table name to make it unique 

249 result = self.name_transformer.normalize_table_name(f"{stream_name}_{hash_json_path([schema] + json_path)}") 

250 else: 

251 # collisions on a nested sub-stream 

252 result = self.name_transformer.normalize_table_name( 

253 get_nested_hashed_table_name(self.name_transformer, schema, json_path, stream_name), False, False 

254 ) 

255 return result

264 def resolve_file_name(self, schema: str, table_name: str) -> str: 

265 """ 

266 We prefer to use file_name = table_name when possible... 

267 

268 When a catalog has ambiguity, we have to fallback and use schema in the file name too 

269 (which might increase a risk of truncate operation and thus collisions that we solve by adding a hash of the full names) 

270 """ 

271 if len(self.simple_file_registry[table_name]) == 1: 

272 # no collisions on file naming 

273 return table_name 

274 else: 

275 max_length = self.name_transformer.get_name_max_length() 

276 # if schema . table fits into the destination, we use this naming convention 

277 if len(schema) + len(table_name) + 1 < max_length: 

278 return f"{schema}_{table_name}" 

279 else: 

280 # we have to make sure our filename is unique, use hash of full name 

281 return self.name_transformer.normalize_table_name(f"{schema}_{table_name}_{hash_name(schema + table_name)}")

283 def get_schema_name(self, schema: str, json_path: List[str], stream_name: str): 

284 """ 

285 Return the schema name from the registry that should be used for this combination of schema/json_path_to_substream

293 def get_table_name(self, schema: str, json_path: List[str], stream_name: str, suffix: str, truncate: bool = False): 

294 """ 

295 Return the table name from the registry that should be used for this combination of schema/json_path_to_substream 

296 """ 

297 key = self.get_registry_key(schema, json_path, stream_name) 

298 if key in self.registry: 

299 table_name = self.registry[key].table_name 

300 else: 

301 raise KeyError(f"Registry does not contain an entry for {schema} {json_path} {stream_name}") 

302 

303 if suffix: 

304 norm_suffix = suffix if not suffix or suffix.startswith("_") else f"_{suffix}" 

305 else: 

306 norm_suffix = "" 

307 

308 conflict = False 

309 conflict_solver = 0 

310 if stream_name in json_path: 

311 conflict = True 

312 conflict_solver = len(json_path) 

313 

314 return self.name_transformer.normalize_table_name(f"{table_name}{norm_suffix}", False, truncate, conflict, conflict_solver)

357def hash_name(input: str) -> str: 

358 h = hashlib.sha1() 

359 h.update(input.encode("utf-8").lower()) 

360 return h.hexdigest()[:3]

363def get_nested_hashed_table_name(name_transformer: DestinationNameTransformer, schema: str, json_path: List[str], child: str) -> str: 

364 """ 

365 In normalization code base, we often have to deal with naming for tables, combining informations from: 

366 - parent table: to denote where a table is extracted from (in case of nesting) 

367 - child table: in case of nesting, the field name or the original stream name 

368 - extra suffix: normalization is done in multiple transformation steps, each may need to generate separate tables, 

369 so we can add a suffix to distinguish the different transformation steps of a pipeline. 

370 - json path: in terms of parent and nested field names in order to reach the table currently being built 

371 

372 All these informations should be included (if possible) in the table naming for the user to (somehow) identify and 

373 recognize what data is available there. 

374 """ 

375 parent = "_".join(json_path[:-1]) 

376 max_length = name_transformer.get_name_max_length() 

377 json_path_hash = hash_json_path([schema] + json_path) 

378 norm_parent = parent if not parent else name_transformer.normalize_table_name(parent, False, False) 

379 norm_child = name_transformer.normalize_table_name(child, False, False) 

380 min_parent_length = min(MINIMUM_PARENT_LENGTH, len(norm_parent)) 

381 

382 # no parent 

383 if not parent: 

384 raise RuntimeError("There is no nested table names without parents") 

385 # if everything fits without truncation, don't truncate anything 

386 elif (len(norm_parent) + len(json_path_hash) + len(norm_child) + 2) < max_length: 

387 return f"{norm_parent}_{json_path_hash}_{norm_child}" 

388 # if everything fits except for the parent, just truncate the parent (still guarantees parent is of length min_parent_length) 

389 elif (min_parent_length + len(json_path_hash) + len(norm_child) + 2) < max_length: 

390 max_parent_length = max_length - len(json_path_hash) - len(norm_child) - 2 

391 return f"{norm_parent[:max_parent_length]}_{json_path_hash}_{norm_child}" 

392 # otherwise first truncate parent to the minimum length and middle truncate the child too 

393 else: 

394 norm_child_max_length = max_length - len(json_path_hash) - 2 - min_parent_length 

395 trunc_norm_child = name_transformer.truncate_identifier_name(norm_child, norm_child_max_length) 

396 return f"{norm_parent[:min_parent_length]}_{json_path_hash}_{trunc_norm_child}"
