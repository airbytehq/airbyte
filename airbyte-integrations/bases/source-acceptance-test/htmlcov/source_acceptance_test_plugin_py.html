53def pytest_generate_tests(metafunc): 

54 """Hook function to customize test discovery and parametrization. 

55 It does two things: 

56 1. skip test class if its name omitted in the config file (or it has no inputs defined) 

57 2. parametrize each test with inputs from config file. 

58 

59 For example config file contains this: 

60 tests: 

61 test_suite1: 

62 - input1: value1 

63 input2: value2 

64 - input1: value3 

65 input2: value4 

66 test_suite2: [] 

67 

68 Hook function will skip test_suite2 and test_suite3, but parametrize test_suite1 with two sets of inputs. 

69 """ 

70 

71 if "inputs" in metafunc.fixturenames: 

72 config_key = metafunc.cls.config_key() 

73 test_name = f"{metafunc.cls.__name__}.{metafunc.function.__name__}" 

74 config = load_config(metafunc.config.getoption("--acceptance-test-config")) 

75 if not hasattr(config.tests, config_key) or not getattr(config.tests, config_key): 

76 pytest.skip(f"Skipping {test_name} because not found in the config") 

77 else: 

78 test_inputs = getattr(config.tests, config_key) 

79 if not test_inputs: 

80 pytest.skip(f"Skipping {test_name} because no inputs provided") 

81 

82 metafunc.parametrize("inputs", test_inputs)

85def pytest_collection_modifyitems(config, items): 

86 """ 

87 Get prepared test items and wrap them with `pytest.mark.timeout(timeout_seconds)` decorator. 

88 

89 `timeout_seconds` may be received either from acceptance test config or `pytest.mark.default_timeout(timeout_seconds)`, 

90 if `timeout_seconds` is not specified in the acceptance test config. 

91 """ 

92 

93 config = load_config(config.getoption("--acceptance-test-config")) 

94 

95 i = 0 

96 packed_items = [] 

97 while i < len(items): 

98 inner_items = [item for item in items if item.originalname == items[i].originalname] 

99 packed_items.append(inner_items) 

100 i += len(inner_items) 

101 

102 for items in packed_items: 

103 test_configs = getattr(config.tests, items[0].cls.config_key()) 

104 for test_config, item in zip(test_configs, items): 

105 default_timeout = item.get_closest_marker("default_timeout") 

106 if test_config.timeout_seconds: 

107 item.add_marker(pytest.mark.timeout(test_config.timeout_seconds)) 

108 elif default_timeout: 

109 item.add_marker(pytest.mark.timeout(*default_timeout.args))
