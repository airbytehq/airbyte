133def primary_keys_for_records(streams, records): 

134 streams_with_primary_key = [stream for stream in streams if stream.stream.source_defined_primary_key] 

135 for stream in streams_with_primary_key: 

136 stream_records = [r for r in records if r.stream == stream.stream.name] 

137 for stream_record in stream_records: 

138 pk_values = {} 

139 for pk_path in stream.stream.source_defined_primary_key: 

140 pk_value = reduce(lambda data, key: data.get(key) if isinstance(data, dict) else None, pk_path, stream_record.data) 

141 pk_values[tuple(pk_path)] = pk_value 

142 

143 yield pk_values, stream_record

245 @staticmethod 

246 def compare_records( 

247 stream_name: str, 

248 actual: List[Dict[str, Any]], 

249 expected: List[Dict[str, Any]], 

250 extra_fields: bool, 

251 exact_order: bool, 

252 extra_records: bool, 

253 detailed_logger: Logger, 

254 ): 

255 """Compare records using combination of restrictions""" 

256 if exact_order: 

257 for r1, r2 in zip(expected, actual): 

258 if r1 is None: 

259 assert extra_records, f"Stream {stream_name}: There are more records than expected, but extra_records is off" 

260 break 

261 if extra_fields: 

262 r2 = TestBasicRead.remove_extra_fields(r2, r1) 

263 assert r1 == r2, f"Stream {stream_name}: Mismatch of record order or values" 

264 else: 

265 expected = set(map(serialize, expected)) 

266 actual = set(map(serialize, actual)) 

267 missing_expected = set(expected) - set(actual) 

268 

269 if missing_expected: 

270 msg = f"Stream {stream_name}: All expected records must be produced" 

271 detailed_logger.info(msg) 

272 detailed_logger.log_json_list(missing_expected) 

273 pytest.fail(msg) 

274 

275 if not extra_records: 

276 extra_actual = set(actual) - set(expected) 

277 if extra_actual: 

278 msg = f"Stream {stream_name}: There are more records than expected, but extra_records is off" 

279 detailed_logger.info(msg) 

280 detailed_logger.log_json_list(extra_actual)
