25import logging 

26import re 

27from collections import defaultdict 

28from typing import List, Mapping

30import pendulum 

31from airbyte_cdk.models import AirbyteRecordMessage, ConfiguredAirbyteCatalog 

32from jsonschema import Draft7Validator, FormatChecker, FormatError, ValidationError

34# fmt: off 

35timestamp_regex = re.compile((r"^\d{4}-\d?\d-\d?\d" # date 

36 r"(\s|T)" # separator 

37 r"\d?\d:\d?\d:\d?\d(.\d+)?" # time 

38 r".*$")) # timezone 

39# fmt: on

41 

42class CustomFormatChecker(FormatChecker): 

43 @staticmethod 

44 def check_datetime(value: str) -> bool: 

45 valid_format = timestamp_regex.match(value) 

46 try: 

47 pendulum.parse(value, strict=False) 

48 except ValueError: 

49 valid_time = False 

50 else: 

51 valid_time = True 

52 return valid_format and valid_time 

53 

54 def check(self, instance, format): 

55 if instance is not None and format == "date-time": 

56 if not self.check_datetime(instance): 

57 raise FormatError(f"{instance} has invalid datetime format") 

58 else: 

59 return super().check(instance, format)

61 

62def verify_records_schema( 

63 records: List[AirbyteRecordMessage], catalog: ConfiguredAirbyteCatalog 

64) -> Mapping[str, Mapping[str, ValidationError]]: 

65 """Check records against their schemas from the catalog, yield error messages. 

66 Only first record with error will be yielded for each stream. 

67 """ 

68 validators = {} 

69 for stream in catalog.streams: 

70 validators[stream.stream.name] = Draft7Validator(stream.stream.json_schema, format_checker=CustomFormatChecker()) 

71 

72 stream_errors = defaultdict(dict) 

73 

74 for record in records: 

75 validator = validators.get(record.stream) 

76 if not validator: 

77 logging.error(f"Record from the {record.stream} stream that is not in the catalog.") 

78 continue 

79 

80 errors = list(validator.iter_errors(record.data)) 

81 for error in errors: 

82 stream_errors[record.stream][str(error.schema_path)] = error 

83 

84 return stream_errors
