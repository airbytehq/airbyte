37class ConnectorRunner: 

38 def __init__(self, image_name: str, volume: Path): 

39 self._client = docker.from_env() 

40 try: 

41 self._image = self._client.images.get(image_name) 

42 except docker.errors.ImageNotFound: 

43 print("Pulling docker image", image_name) 

44 self._image = self._client.images.pull(image_name) 

45 print("Pulling completed") 

46 self._runs = 0 

47 self._volume_base = volume 

48 

49 @property 

50 def output_folder(self) -> Path: 

51 return self._volume_base / f"run_{self._runs}" / "output" 

52 

53 @property 

54 def input_folder(self) -> Path: 

55 return self._volume_base / f"run_{self._runs}" / "input" 

56 

57 def _prepare_volumes(self, config: Optional[Mapping], state: Optional[Mapping], catalog: Optional[ConfiguredAirbyteCatalog]): 

58 self.input_folder.mkdir(parents=True) 

59 self.output_folder.mkdir(parents=True) 

60 

61 if config: 

62 with open(str(self.input_folder / "tap_config.json"), "w") as outfile: 

63 json.dump(dict(config), outfile) 

64 

65 if state: 

66 with open(str(self.input_folder / "state.json"), "w") as outfile: 

67 json.dump(dict(state), outfile) 

68 

69 if catalog: 

70 with open(str(self.input_folder / "catalog.json"), "w") as outfile: 

71 outfile.write(catalog.json()) 

72 

73 volumes = { 

74 str(self.input_folder): { 

75 "bind": "/data", 

76 # "mode": "ro", 

77 }, 

78 str(self.output_folder): { 

79 "bind": "/local", 

80 "mode": "rw", 

81 }, 

82 } 

83 return volumes

85 def call_spec(self, **kwargs) -> List[AirbyteMessage]: 

86 cmd = "spec" 

87 output = list(self.run(cmd=cmd, **kwargs)) 

88 return output 

89 

90 def call_check(self, config, **kwargs) -> List[AirbyteMessage]: 

91 cmd = "check --config tap_config.json" 

92 output = list(self.run(cmd=cmd, config=config, **kwargs)) 

93 return output 

94 

95 def call_discover(self, config, **kwargs) -> List[AirbyteMessage]: 

96 cmd = "discover --config tap_config.json" 

97 output = list(self.run(cmd=cmd, config=config, **kwargs)) 

98 return output 

99 

100 def call_read(self, config, catalog, **kwargs) -> List[AirbyteMessage]: 

101 cmd = "read --config tap_config.json --catalog catalog.json" 

102 output = list(self.run(cmd=cmd, config=config, catalog=catalog, **kwargs)) 

103 return output 

104 

105 def call_read_with_state(self, config, catalog, state, **kwargs) -> List[AirbyteMessage]: 

106 cmd = "read --config tap_config.json --catalog catalog.json --state state.json" 

107 output = list(self.run(cmd=cmd, config=config, catalog=catalog, state=state, **kwargs)) 

108 return output 

109 

110 def run(self, cmd, config=None, state=None, catalog=None, **kwargs) -> Iterable[AirbyteMessage]: 

111 self._runs += 1 

112 volumes = self._prepare_volumes(config, state, catalog) 

113 logging.info("Docker run: \n%s\ninput: %s\noutput: %s", cmd, self.input_folder, self.output_folder) 

114 try: 

115 logs = self._client.containers.run( 

116 image=self._image, command=cmd, working_dir="/data", volumes=volumes, network="host", stdout=True, stderr=True, **kwargs 

117 ) 

118 except ContainerError as err: 

119 # beautify error from container 

120 patched_error = ContainerError( 

121 container=err.container, exit_status=err.exit_status, command=err.command, image=err.image, stderr=err.stderr.decode() 

122 ) 

123 raise patched_error from None # get rid of any previous exception stack 

124 

125 with open(str(self.output_folder / "raw"), "wb+") as f: 

126 f.write(logs) 

127 

128 for line in logs.decode("utf-8").splitlines(): 

129 try: 

130 yield AirbyteMessage.parse_raw(line) 

131 except ValidationError as exc: 

132 logging.warning("Unable to parse connector's output %s", exc)
